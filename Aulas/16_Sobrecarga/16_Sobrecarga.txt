  int num, den;

#include <iostream>
#include "Fraction.h"

  const double commonDen = den * f.den;
  return (num * f.den)/commonDen < (f.num * den)/commonDen;

  os << dt.num << "/" << dt.den;

// Fraction.h

struct Fraction {
  int num, den;
  bool operator < (const Fraction& f) const;
  inline bool operator > (const Fraction& rhs) const {
    return rhs < *this;
  }
  inline bool operator <= (const Fraction& rhs) const {
    return !(*this > rhs);
  }
  inline bool operator >= (const Fraction& rhs) const {

    if (y == 0) {
      return x;
    } else {
      if (y > x) {
        return gcd(y, x);
      } else {
        int r = x / y;
        return gcd(y, x - r * y);
